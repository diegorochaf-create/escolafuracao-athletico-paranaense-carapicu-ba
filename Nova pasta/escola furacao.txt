    <!--
        PARA FAZER O FORMULÁRIO FUNCIONAR, SIGA ESTES PASSOS:

        1. ABRA SUA PLANILHA GOOGLE:
           Acesse a planilha: https://docs.google.com/spreadsheets/d/10s6btljAey-A9mKJEoFuIQDHyZtjS93Gra6Yjhsd8ac/edit

        2. ABRA O EDITOR DE SCRIPT:
           No menu superior, clique em "Extensões" > "Apps Script".

        3. COLE O CÓDIGO ABAIXO:
           Apague todo o código que estiver no editor e cole o código que está abaixo destas instruções.
           Este código irá receber os dados do formulário e adicioná-los à sua planilha.

           // CÓDIGO DO GOOGLE APPS SCRIPT (PARA COLAR LÁ)
           function doPost(e) {
             try {
               var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
               
               // Pega os cabeçalhos da primeira linha
               var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
               var newRow = [];
               
               // Adiciona um timestamp na primeira coluna
               newRow.push(new Date());

               // Mapeia os dados do formulário para as colunas corretas
               for (var i = 1; i < headers.length; i++) {
                 var header = headers[i];
                 newRow.push(e.parameter[header] || ""); // Usa "" se o campo não existir
               }
               
               // Adiciona a nova linha
               sheet.appendRow(newRow);
               
               return ContentService.createTextOutput(JSON.stringify({ "result": "success" })).setMimeType(ContentService.MimeType.JSON);
             } catch (error) {
               return ContentService.createTextOutput(JSON.stringify({ "result": "error", "error": error.toString() })).setMimeType(ContentService.MimeType.JSON);
             }
           }


        4. FAÇA A IMPLANTAÇÃO (DEPLOY):
           a. No canto superior direito do editor de script, clique no botão azul "Implantar".
           b. Selecione "Nova implantação".
           c. Clique no ícone de engrenagem (??) ao lado de "Selecione o tipo" e escolha "App da Web".
           d. Em "Descrição", coloque algo como "Formulário de Contato Site Furacão".
           e. Em "Executar como", deixe "Eu (seu.email@gmail.com)".
           f. Em "Quem pode acessar", selecione "Qualquer pessoa". **ISSO É MUITO IMPORTANTE!**
           g. Clique em "Implantar".

        5. AUTORIZE O SCRIPT:
           O Google pedirá permissão para que o script acesse sua planilha. Clique em "Autorizar acesso" e siga as instruções (pode ser necessário clicar em "Avançado" e depois em "Acessar... (não seguro)", mas não se preocupe, é seguro pois o script é seu).

        6. COPIE A URL DO APP DA WEB:
           Após a implantação, uma janela aparecerá com a "URL do app da Web". Copie essa URL.

        7. COLE A URL NO CÓDIGO ABAIXO:
           No código JavaScript desta página (logo abaixo), substitua a frase 'COLE_A_URL_DO_SEU_SCRIPT_AQUI' pela URL que você acabou de copiar.
    -->
    <script>
        const form = document.getElementById('inscricao-form');
        const submitButton = document.getElementById('submit-button');
        const formMessage = document.getElementById('form-message');

        // ??? URL DO SCRIPT DO GOOGLE SHEETS ???
        const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbz-sEz87aD1fUA3XOFOMDeXaHdbxc7YpagGrV6CE82dnmH-_M-m8U4jUJ07QjGlAmLwkw/exec'; 
        // ??? URL DO SCRIPT DO GOOGLE SHEETS ???

        form.addEventListener('submit', e => {
            e.preventDefault();
            
            if (SCRIPT_URL === 'COLE_A_URL_DO_SEU_SCRIPT_AQUI') {
                formMessage.innerHTML = '<p class="text-red-500 font-bold">Erro: A URL do script do Google Sheets não foi configurada.</p>';
                return;
            }

            submitButton.disabled = true;
            submitButton.innerHTML = 'Enviando...';
            formMessage.innerHTML = '';

            const formData = new FormData(form);
            
            fetch(SCRIPT_URL, {
                method: 'POST',
                body: formData
            })
            .then(res => res.json())
            .then(data => {
                if (data.result === 'success') {
                    formMessage.innerHTML = '<p class="text-green-600 font-bold">Inscrição enviada com sucesso! Entraremos em contato em breve.</p>';
                    form.reset();
                } else {
                    throw new Error(data.error || 'Ocorreu um erro desconhecido.');
                }
            })
            .catch(error => {
                console.error('Error!', error.message);
                formMessage.innerHTML = `<p class="text-red-500 font-bold">Ocorreu um erro ao enviar a inscrição. Tente novamente. Detalhe: ${error.message}</p>`;
            })
            .finally(() => {
                submitButton.disabled = false;
                submitButton.innerHTML = 'Enviar Inscrição';
            });
        });
    </script>
